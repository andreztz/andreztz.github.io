Conversão datetime/time

from datetime import datetime
import calendar
import time

#-------------------------------------------------
# conversão para string
#-------------------------------------------------
# datetime objeto para string
date_obj = datetime(2008, 11, 10, 17, 53, 59)
date_str = date_obj.strftime("%Y-%m-%d %H:%M:%S")
print(date_str)

# time tuple para string
time_tuple = (2008, 11, 12, 13, 51, 18, 2, 317, 0)
date_str = time.strftime("%Y-%m-%d %H:%M:%S", time_tuple)
print(date_str)

#-------------------------------------------------
# conversão para objetos datetime
#-------------------------------------------------
# time tuple para objetos datetime
time_tuple = (2008, 11, 12, 13, 51, 18, 2, 317, 0)
date_obj = datetime(*time_tuple[0:6])
print(date_obj)

# data string para objeto datetime
date_str = "2008-11-10 17:53:59"
date_obj = datetime.strptime(date_str, "%Y-%m-%d %H:%M:%S")
print(date_obj)

# timestamp para objeto datetime em local time
timestamp = 1226527167.595983
date_obj = datetime.fromtimestamp(timestamp)
print(date_obj)

# timestamp para objeto datetime em UTC
timestamp = 1226527167.595983
date_obj = datetime.utcfromtimestamp(timestamp)
print(date_obj)

#-------------------------------------------------
# conversão para time tuples
#-------------------------------------------------
# objetos datetime para time tuple
date_obj = datetime(2008, 11, 10, 17, 53, 59)
time_tuple = date_obj.timetuple()
print(time_tuple)

# string para time tuple
date_str = "2008-11-10 17:53:59"
time_tuple = time.strptime(date_str, "%Y-%m-%d %H:%M:%S")
print(time_tuple)

# timestamp para time tuple in UTC
timestamp = 1226527167.595983
time_tuple = time.gmtime(timestamp)
print(time_tuple)

# timestamp para time tuple em local time
timestamp = 1226527167.595983
time_tuple = time.localtime(timestamp)
print(time_tuple)

#-------------------------------------------------
# conversão para timestamps
#-------------------------------------------------
# time tuple em local time para timestamp
time_tuple = (2008, 11, 12, 13, 59, 27, 2, 317, 0)
timestamp = time.mktime(time_tuple)
print(timestamp)

# time tuple em utc time para timestamp
time_tuple_utc = (2008, 11, 12, 13, 59, 27, 2, 317, 0)
timestamp_utc = calendar.timegm(time_tuple_utc)
print(timestamp_utc)
